120  virtual void subscribe(Topic&& topic, EventSlot&& slot,
121  AsyncTask<Subscription>&& handler) override
122  {
123  using std::move;
124  SubscriptionRecord rec = {topic.uri(), move(slot), handler.executor()};
125 
126  auto kv = topics_.find(rec.topicUri);
127  if (kv == topics_.end())
128  {
129  auto self = this->shared_from_this();
130  this->request(
131  topic.message({}),
132  [this, self, CPPWAMP_MVCAP(rec), CPPWAMP_MVCAP(handler)]
133  (std::error_code ec, Message reply)
134  {
135  if (checkReply(WampMsgType::subscribed, ec, reply,
136  SessionErrc::subscribeError, handler))
137  {
138  const auto& subMsg =
139  message_cast<SubscribedMessage>(reply);
140  auto subId = subMsg.subscriptionId();
141  auto slotId = nextSlotId();
142  Subscription sub(self, subId, slotId, {});
143  topics_.emplace(rec.topicUri, subId);
144  readership_[subId][slotId] = move(rec);
145  std::move(handler)(std::move(sub));
146  }
147  });
148  }
149  else
150  {
151  auto subId = kv->second;
152  auto slotId = nextSlotId();
153  Subscription sub{this->shared_from_this(), subId, slotId, {}};
154  readership_[subId][slotId] = move(rec);
155  std::move(handler)(move(sub));
156  }
157  }

159  virtual void unsubscribe(const Subscription& sub) override
160  {
161  auto kv = readership_.find(sub.id());
162  if (kv != readership_.end())
163  {
164  auto& localSubs = kv->second;
165  if (!localSubs.empty())
166  {
167  auto subKv = localSubs.find(sub.slotId({}));
168  if (subKv != localSubs.end())
169  {
170  if (localSubs.size() == 1u)
171  topics_.erase(subKv->second.topicUri);
172 
173  localSubs.erase(subKv);
174  if (localSubs.empty())
175  {
176  readership_.erase(kv);
177  sendUnsubscribe(sub.id());
178  }
179  }
180  }
181  }

189  if (kv != readership_.end())

194  auto subKv = localSubs.find(sub.slotId({}));

312  {
313  using std::move;

316  auto opts = RequestOptions().withProgressiveResponse(

318  auto self = this->shared_from_this();

325  SessionErrc::callError, handler, errorPtr))

333  timeoutScheduler_->add(rpc.callerTimeout(), requestId);

534  void onEvent(Message&& msg)
535  {
536  auto& eventMsg = message_cast<EventMessage>(msg);
537  auto kv = readership_.find(eventMsg.subscriptionId());
538  if (kv != readership_.end())
539  {
540  const auto& localSubs = kv->second;
541  assert(!localSubs.empty());
542  Event event({}, localSubs.begin()->second.executor,
543  std::move(eventMsg));
544  for (const auto& subKv: localSubs)
545  dispatchEvent(subKv.second, event);
546  }
547  else if (warningHandler_)
548  {
549  std::ostringstream oss;
550  oss << "Received an EVENT that is not subscribed to "
551  "(with subId=" << eventMsg.subscriptionId()
552  << " pubId=" << eventMsg.publicationId() << ")";
553  warn(oss.str());
554  }
555  }

557  void dispatchEvent(const SubscriptionRecord& sub, const Event& event)
558  {
559  auto self = this->shared_from_this();
560  const auto& slot = sub.slot;
561  boost::asio::post(sub.executor, [this, self, slot, event]()
562  {
563  // Copy the subscription and publication IDs before the Event
564  // object gets moved away.
565  auto subId = event.subId();
566  auto pubId = event.pubId();
567 
568  /* The catch clauses are to prevent the publisher crashing
569  subscribers when it passes arguments having incorrect type. */
570  try
571  {
572  slot(std::move(event));
573  }
574  catch (const Error& e)
575  {
576  if (warningHandler_)
577  warnEventError(e, subId, pubId);
578  }
579  catch (const error::BadType& e)
580  {
581  if (warningHandler_)
582  warnEventError(Error(e), subId, pubId);
583  }
584  });
585  }

598  void onInvocation(Message&& msg)
599  {
600  auto& invMsg = message_cast<InvocationMessage>(msg);
601  auto requestId = invMsg.requestId();
602  auto regId = invMsg.registrationId();
603 
604  auto kv = registry_.find(regId);
605  if (kv != registry_.end())
606  {
607  auto self = this->shared_from_this();
608  const RegistrationRecord& rec = kv->second;
609  Invocation inv({}, self, rec.executor, std::move(invMsg));
610  pendingInvocations_[requestId] = regId;
611  dispatchRpcRequest(rec.executor, rec.callSlot, std::move(inv));
612  }
613  else
614  {
615  this->sendError(WampMsgType::invocation, requestId,
616  Error("wamp.error.no_such_procedure"));
617  }
618  }

620  void onInterrupt(Message&& msg)
621  {
622  auto& interruptMsg = message_cast<InterruptMessage>(msg);
623  auto found = pendingInvocations_.find(interruptMsg.requestId());
624  if (found != pendingInvocations_.end())
625  {
626  auto registrationId = found->second;
627  pendingInvocations_.erase(found);
628  auto kv = registry_.find(registrationId);
629  if ((kv != registry_.end()) &&
630  (kv->second.interruptSlot != nullptr))
631  {
632  auto self = this->shared_from_this();
633  const RegistrationRecord& rec = kv->second;
634  Interruption intr({}, self, rec.executor,
635  std::move(interruptMsg));
636  dispatchRpcRequest(rec.executor, rec.interruptSlot,
637  std::move(intr));
638  }
639  }

642  template <typename TSlot, typename TRequest>
643  void dispatchRpcRequest(AnyExecutor exec, const TSlot& slot,
644  TRequest&& request)
645  {
646  auto self = this->shared_from_this();
647  boost::asio::post(exec, [this, self, slot, CPPWAMP_MVCAP(request)]()
648  {
649  // Copy the request ID before the request object gets moved away.
650  auto requestId = request.requestId();
651 
652  try
653  {
654  Outcome outcome(slot(std::move(request)));
655  switch (outcome.type())
656  {
657  case Outcome::Type::deferred:
658  // Do nothing
659  break;
660 
661  case Outcome::Type::result:
662  yield(requestId, std::move(outcome).asResult());
663  break;
664 
665  case Outcome::Type::error:
666  yield(requestId, std::move(outcome).asError());
667  break;
668 
669  default:
670  assert(false && "unexpected Outcome::Type");
671  }
672  }
673  catch (Error& error)
674  {
675  yield(requestId, std::move(error));
676  }
677  catch (const error::BadType& e)
678  {
679  // Forward Variant conversion exceptions as ERROR messages.
680  yield(requestId, Error(e));
681  }
682  });
683  }

685  template <typename THandler>
686  bool checkError(std::error_code ec, THandler&& handler)
687  {
688  if (ec)
689  std::forward<THandler>(handler)(ec);
690  return !ec;
691  }
692 
693  template <typename THandler>
694  bool checkReply(WampMsgType type, std::error_code ec, Message& reply,
695  SessionErrc defaultErrc, THandler&& handler,
696  Error* errorPtr = nullptr)
697  {
698  bool success = checkError(ec, handler);
699  if (success)
700  {
701  if (reply.type() == WampMsgType::error)
702  {
703  success = false;
704  auto& errMsg = message_cast<ErrorMessage>(reply);
705  const auto& uri = errMsg.reasonUri();
706  auto errc = lookupWampErrorUri(uri, defaultErrc);
707 
708  std::ostringstream oss;
709  oss << "with URI=" << uri;
710  if (!errMsg.args().empty())
711  oss << ", Args=" << errMsg.args();
712  if (!errMsg.kwargs().empty())
713  oss << ", ArgsKv=" << errMsg.kwargs();
714 
715  if (errorPtr != nullptr)
716  *errorPtr = Error({}, std::move(errMsg));
717 
718  using ResultType = typename ResultTypeOfHandler<
719  typename std::decay<THandler>::type >::Type;
720  ResultType result(make_error_code(errc), oss.str());
721  std::forward<THandler>(handler)(std::move(result));
722  }
723  else
724  assert((reply.type() == type) && "Unexpected WAMP message type");
725  }
726  return success;
727  }
