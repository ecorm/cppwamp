32 template <typename TEstablisher,
33  template<typename> class TTransport = AsioTransport>
34 class AsioEndpoint
35 {
36 public:
37  using Establisher = TEstablisher;
38  using Socket = typename Establisher::Socket;
39  using Transport = TTransport<Socket>;
40  using TransportPtr = std::shared_ptr<Transport>;
41  using Handler = std::function<void (std::error_code, int codecId,
42  TransportPtr)>;
43 
44  explicit AsioEndpoint(Establisher&& est)
45  : executor_(est.executor()),
46  est_(std::move(est)),
47  handshake_(0)
48  {}
49 
50  void establish(Handler&& handler)
51  {
52  handler_ = std::move(handler);
53  try
54  {
55  // The Connector that initiated 'establish' must keep this object
56  // alive until completion.
57  est_.establish( [this](AsioErrorCode ec, SocketPtr&& socket)
58  {
59  if (check(ec))
60  {
61  socket_ = std::move(socket);
62  onEstablished();
63  }
64  });
65  }
66  catch (const boost::system::system_error& e)
67  {
68  check(e.code());
69  }
70  }
71 
72  void cancel()
73  {
74  if (socket_)
75  socket_->close();
76  else
77  est_.cancel();
78  }
79 
80 protected:
81  using Handshake = internal::RawsockHandshake;
82  using SocketPtr = typename Establisher::SocketPtr;
83 
84  void sendHandshake(Handshake hs)
85  {
86  handshake_ = hs.toBigEndian();
87  using boost::asio::buffer;
88  using boost::asio::async_write;
89  async_write(*socket_, buffer(&handshake_, sizeof(handshake_)),
90  [this, hs](AsioErrorCode ec, size_t)
91  {
92  if (check(ec))
93  onHandshakeSent(hs);
94  });
95  }
96 
97  void receiveHandshake()
98  {
99  handshake_ = 0;
100  using boost::asio::buffer;
101  using boost::asio::async_read;
102  async_read(*socket_, buffer(&handshake_, sizeof(handshake_)),
103  [this](AsioErrorCode ec, size_t)
104  {
105  if (check(ec))
106  onHandshakeReceived(Handshake::fromBigEndian(handshake_));
107  });
108  }
109 
110  void complete(int codecId, size_t maxTxLength, size_t maxRxLength)
111  {
112  auto transport = Transport::create(std::move(socket_), maxTxLength,
113  maxRxLength);
114  std::error_code ec = make_error_code(TransportErrc::success);
115  post(std::bind(handler_, ec, codecId, std::move(transport)));
116  socket_.reset();
117  handler_ = nullptr;
118  }
119 
120  void fail(RawsockErrc errc)
121  {
122  socket_.reset();
123  post(std::bind(handler_, make_error_code(errc), 0, nullptr));
124  handler_ = nullptr;
125  }
126 
127  bool check(AsioErrorCode asioEc)
128  {
129  if (asioEc)
130  {
131  auto ec = make_error_code(static_cast<std::errc>(asioEc.value()));
132  socket_.reset();
133  post(std::bind(handler_, ec, 0, nullptr));
134  handler_ = nullptr;
135  }
136  return !asioEc;
137  }
138 
139  virtual void onEstablished() = 0;
140 
141  virtual void onHandshakeReceived(Handshake hs) = 0;
142 
143  virtual void onHandshakeSent(Handshake hs) = 0;
144 
145 private:
146  template <typename TFunction>
147  void post(TFunction&& fn)
148  {
149  boost::asio::post(executor_, std::forward<TFunction>(fn));
150  }
151 
152  AnyExecutor executor_;
153  SocketPtr socket_;
154  Handler handler_;
155  Establisher est_;
156  uint32_t handshake_;
157 
158  // Grant friendship to mock objects used in testing
159  friend class wamp::FakeMsgTypeAsioConnector;
160  friend class wamp::FakeMsgTypeAsioListener;
161 };
