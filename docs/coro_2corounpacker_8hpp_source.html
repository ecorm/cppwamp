279 //------------------------------------------------------------------------------
280 template <typename... A>
281 struct UnpackedCoroArgGetter
282 {
283  template <int N>
284  static NthTypeOf<N, A...> get(const Array& args)
285  {
286  using TargetType = NthTypeOf<N, A...>;
287  try
288  {
289  return args.at(N).to<TargetType>();
290  }
291  catch(const error::Conversion& e)
292  {
293  std::ostringstream oss;
294  oss << "Type " << typeNameOf(args.at(N))
295  << " at arg index " << N
296  << " is not convertible to the RPC's target type";
297  throw UnpackCoroError().withArgs(oss.str(), e.what());
298  }
299  }
300 };

323  << " args, but only got " << event.args().size();
324  throw internal::UnpackCoroError().withArgs(oss.str());
325  }

330  invoke(std::move(event), Seq());

333 //------------------------------------------------------------------------------
334 template <typename S, typename... A>
335 template <int ...Seq>
336 void CoroEventUnpacker<S,A...>::invoke(Event&& event,
337  internal::IntegerSequence<Seq...>)
338 {
339  auto slot = slot_;
340  boost::asio::spawn(event.executor(), [slot, event](Yield yield)
341  {
342  Array args = event.args();
343  using Getter = internal::UnpackedCoroArgGetter<A...>;
344  try
345  {
346  (*slot)(std::move(event), Getter::template get<Seq>(args)...,
347  yield);
348  }
349  catch (const error::BadType& e)
350  {
351  /* Do nothing. This is to prevent the publisher crashing
352  subscribers when it passes Variant objects having
353  incorrect schema. */
354  }
355  });
356 }

460 //------------------------------------------------------------------------------
461 template <typename S, typename... A>
462 template <int ...Seq>
463 void CoroInvocationUnpacker<S,A...>::invoke(Invocation&& inv,
464  internal::IntegerSequence<Seq...>)
465 {
466  auto slot = slot_;
467  boost::asio::spawn(inv.executor(), [slot, inv](Yield yield)
468  {
469  try
470  {
471  Array args = inv.args();
472  using Getter = internal::UnpackedCoroArgGetter<A...>;
473  Outcome outcome = (*slot)(std::move(inv),
474  Getter::template get<Seq>(args)...,
475  yield);
476 
477  switch (outcome.type())
478  {
479  case Outcome::Type::deferred:
480  // Do nothing
481  break;
482 
483  case Outcome::Type::result:
484  inv.yield(std::move(outcome).asResult());
485  break;
486 
487  case Outcome::Type::error:
488  inv.yield(std::move(outcome).asError());
489  break;
490 
491  default:
492  assert(false && "Unexpected wamp::Outcome::Type enumerator");
493  }
494 
495  }
496  catch (const Error& e)
497  {
498  inv.yield(e);
499  }
500  catch (const error::BadType& e)
501  {
502  // Forward Variant conversion exceptions as ERROR messages.
503  inv.yield(Error(e));
504  }
505  });
506 }
