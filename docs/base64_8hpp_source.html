26  using Byte = uint8_t;
27  using ByteVec = std::vector<uint8_t>;

29  template <typename TBuffer>
30  static void encode(const ByteVec& data, TBuffer& out)
31  {
32  Quad quad;
33  Byte sextet;
34  auto byte = data.cbegin();
35  while (byte != data.cend())
36  {
37  quad[0] = charFromSextet( (*byte >> 2) & 0x3f );
38  sextet = (*byte << 4) & 0x30;
39  ++byte;
40 
41  if (byte != data.cend())
42  {
43  sextet |= (*byte >> 4) & 0x0f;
44  quad[1] = charFromSextet(sextet);
45  sextet = (*byte << 2) & 0x3c;
46  ++byte;
47 
48  if (byte != data.cend())
49  {
50  sextet |= (*byte >> 6) & 0x03;
51  quad[2] = charFromSextet(sextet);
52  quad[3] = charFromSextet( *byte & 0x3f );
53  ++byte;
54  }
55  else
56  {
57  quad[2] = charFromSextet(sextet);
58  quad[3] = pad;
59  }
60  }
61  else
62  {
63  quad[1] = charFromSextet(sextet);
64  quad[2] = pad;
65  quad[3] = pad;
66  }
67 
68  out.write(quad.data(), quad.size());
69  }
70  }

72  static void decode(const char* str, size_t length, ByteVec& data)
73  {
74  if (length == 0)
75  return;
76  if (length % 4 != 0)
77  throw error::Decode("Invalid JSON Base64 payload length");
78 
79  Triplet triplet;
80  const char* quad = str;
81  auto end = str + length - 4;
82  assert(end >= str);
83  for (; quad < end; quad += 4)
84  {
85  triplet = tripletFromQuad(quad, false);
86  data.insert(data.end(), triplet.begin(), triplet.end());
87  }
88 
89  assert((quad + 4) == (str + length));
90 
91  unsigned lastTripletCount = 1;
92  triplet = tripletFromQuad(quad, true);
93  if (quad[0] == pad || quad[1] == pad)
94  throw error::Decode("Invalid JSON Base64 padding");
95 
96  if (quad[2] == pad)
97  {
98  if (quad[3] != pad)
99  throw error::Decode("Invalid JSON Base64 padding");
100  }
101  else
102  lastTripletCount = (quad[3] == pad) ? 2 : 3;
103 
104  data.insert(data.end(), triplet.begin(),
105  triplet.begin() + lastTripletCount);
106  }

108 private:

110  using Quad = std::array<char, 4>;

114  static char charFromSextet(uint8_t sextet)
115  {
116  static const char alphabet[] =
117  {
118  'A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', // 0-7
119  'I', 'J', 'K', 'L', 'M', 'N', 'O', 'P', // 8-15
120  'Q', 'R', 'S', 'T', 'U', 'V', 'W', 'X', // 16-23
121  'Y', 'Z', 'a', 'b', 'c', 'd', 'e', 'f', // 24-31
122  'g', 'h', 'i', 'j', 'k', 'l', 'm', 'n', // 32-39
123  'o', 'p', 'q', 'r', 's', 't', 'u', 'v', // 40-47
124  'w', 'x', 'y', 'z', '0', '1', '2', '3', // 48-55
125  '4', '5', '6', '7', '8', '9', '+', '/' // 56-63
126  };
127  assert(sextet < sizeof(alphabet));
128  return alphabet[sextet];
129  }

131  static Triplet tripletFromQuad(const char* quad, bool padAllowed)
132  {
133  std::array<Byte, 4> sextet;
134  for (unsigned i=0; i<4; ++i)
135  sextet[i] = sextetFromChar(quad[i], padAllowed);
136  Triplet triplet;
137  triplet[0] = ((sextet[0] << 2) & 0xfc) | ((sextet[1] >> 4) & 0x03);
138  triplet[1] = ((sextet[1] << 4) & 0xf0) | ((sextet[2] >> 2) & 0x0f);
139  triplet[2] = ((sextet[2] << 6) & 0xc0) | ((sextet[3] ) & 0x3f);
140  return triplet;
141  }
