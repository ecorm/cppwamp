22 // is to be posted.
23 //------------------------------------------------------------------------------
24 template <typename TResult>

32  AsyncTask(AnyExecutor exec, AsyncHandler<TResult> handler)
33  : executor_(exec),

35  {}

59  return *this;
60  }

62  explicit operator bool() const {return executor_ != nullptr;}

67 
68  void operator()(AsyncResult<ValueType> result) const &
69  {
70  assert(executor_ && "Invoking uninitialized AsyncTask");
71  boost::asio::post(executor_, std::bind(handler_, std::move(result)));
72  }
73 
74  void operator()(AsyncResult<ValueType> result) &&
75  {

77  boost::asio::post(executor_, std::bind(std::move(handler_),
78  std::move(result)));
79  }

81 private:
