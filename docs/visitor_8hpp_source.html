86 template <typename V, typename T>
87 ResultTypeOf<V> apply(V&& visitor, T&& variant)
88 {
89  using I = decltype(variant.typeId());
90  switch (variant.typeId())
91  {
92  case I::null: return visitor(variant.template as<I::null>());
93  case I::boolean: return visitor(variant.template as<I::boolean>());
94  case I::integer: return visitor(variant.template as<I::integer>());
95  case I::uint: return visitor(variant.template as<I::uint>());
96  case I::real: return visitor(variant.template as<I::real>());
97  case I::string: return visitor(variant.template as<I::string>());
98  case I::blob: return visitor(variant.template as<I::blob>());
99  case I::array: return visitor(variant.template as<I::array>());
100  case I::object: return visitor(variant.template as<I::object>());
101  default: assert(false);
102  }
103 
104  // Unreachable. Return null case to silence warning.
105  return visitor(variant.template as<I::null>());
106 }
