63  void start()

89  void adjourn(Reason reason, Handler handler)
90  {
91  assert(state_ == State::established);
92 
93  state_ = State::shuttingDown;
94  using std::move;
95  Message msg = { WampMsgType::goodbye,
96  {0u, move(reason.options({})), move(reason.uri({}))} };
97  auto self = this->shared_from_this();
98  request(msg,
99  [this, self, handler](std::error_code ec, Message reply)
100  {
101  if (!ec)
102  {
103  state_ = State::closed;
104  abortPending(make_error_code(SessionErrc::sessionEnded));
105  post(handler, make_error_code(ProtocolErrc::success),
106  move(reply));
107  }
108  else
109  post(handler, ec, Message());
110  });

125  void sendError(WampMsgType reqType, RequestId reqId, Error failure)
126  {
127  using std::move;
128 
129  if (!failure.kwargs().empty())
130  {
131  auto& f = errorWithKwargsMsg_.fields;
132  f.at(1) = static_cast<Int>(reqType);
133  f.at(2) = reqId;
134  f.at(3) = move(failure.options({}));
135  f.at(4) = move(failure.reason({}));
136  f.at(5) = move(failure.args({}));
137  f.at(6) = move(failure.kwargs({}));
138  send(errorWithKwargsMsg_);
139  }
140  else if (!failure.args().empty())
141  {
142  auto& f = errorWithArgsMsg_.fields;
143  f.at(1) = static_cast<Int>(reqType);
144  f.at(2) = reqId;
145  f.at(3) = move(failure.options({}));
146  f.at(4) = move(failure.reason({}));
147  f.at(5) = move(failure.args({}));
148  send(errorWithArgsMsg_);
149  }
150  else
151  {
152  auto& f = errorMsg_.fields;
153  f.at(1) = static_cast<Int>(reqType);
154  f.at(2) = reqId;
155  f.at(3) = move(failure.options({}));
156  f.at(4) = move(failure.reason({}));
157  send(errorMsg_);
158  }

190  RequestId sendMessage(Message& msg, Handler&& handler = nullptr)
191  {
192  assert(msg.type != WampMsgType::none);
193 
194  msg.fields.at(0) = static_cast<Int>(msg.type);
195 
196  RequestId requestId = 0;
197  auto idPos = msg.traits().idPosition;
198  if (idPos > 0)
199  {
200  requestId = nextRequestId();
201  msg.fields.at(idPos) = requestId;
202  }
203 
204  trace(msg, true);
205 
206  auto buf = newBuffer(msg);
207  if (buf->length() > transport_->maxSendLength())
208  throw error::Failure(make_error_code(TransportErrc::badTxLength));
209 
210  if (handler)
211  requestMap_[msg.requestKey()] = std::move(handler);
212 
213  transport_->send(std::move(buf));
214 
215  return requestId;
216  }

218  RequestId nextRequestId()

232  void onTransportRx(Buffer buf)
233  {
234  if (state_ == State::establishing ||
235  state_ == State::authenticating ||
236  state_ == State::established ||
237  state_ == State::shuttingDown)
238  {
239  Variant v;
240  if (checkError(decode(buf, v)) &&
241  check(v.is<Array>(), ProtocolErrc::badSchema))
242  {
243  std::error_code ec;
244  auto msg = Message::parse(std::move(v.as<Array>()), ec);
245  if ( checkError(ec) && checkValidMsg(msg.type) )
246  {
247  trace(msg, false);
248  processMessage(std::move(msg));
249  }
250  }
251  }
252  }

254  std::error_code decode(const Buffer& buf, Variant& v)
255  {
256  auto result = make_error_code(ProtocolErrc::success);
257  try
258  {
259  Codec::decodeBuffer(*buf, v);
260  }
261  catch(const error::Decode& e)
262  {
263  result = make_error_code(ProtocolErrc::badDecode);
264  }
265  return result;
266  }

268  void processMessage(Message&& msg)
269  {
270  if (msg.repliesTo() != WampMsgType::none)
271  {
272  processWampReply( RequestKey(msg.repliesTo(), msg.requestId()),
273  std::move(msg) );
274  }
275  else switch(msg.type)
276  {
277  case WampMsgType::hello:
278  processHello(std::move(msg));
279  break;
280 
281  case WampMsgType::challenge:
282  processChallenge(std::move(msg));
283  break;
284 
285  case WampMsgType::welcome:
286  processWelcome(std::move(msg));
287  break;
288 
289  case WampMsgType::abort:
290  processAbort(std::move(msg));
291  break;
292 
293  case WampMsgType::goodbye:
294  processGoodbye(std::move(msg));
295  break;
296 
297  case WampMsgType::error:
298  processWampReply(msg.requestKey(), std::move(msg));
299  break;
300 
301  default:
302  // Role-specific unsolicited messages. Ignore them if we're
303  // shutting down.
304  if (state_ != State::shuttingDown)
305  {
306  auto self = this->shared_from_this();
307  post(std::bind(&Peer::onInbound, self, std::move(msg)));
308  }
309  break;
310  }
311  }

313  void processWampReply(const RequestKey& key, Message&& msg)
314  {
315  auto kv = requestMap_.find(key);
316  if (kv != requestMap_.end())
317  {
318  auto handler = std::move(kv->second);
319  requestMap_.erase(kv);
320  post(handler, make_error_code(ProtocolErrc::success),
321  std::move(msg));
322  }
323  }

359  void processGoodbye(Message&& msg)
360  {
361  if (state_ == State::shuttingDown)
362  {
363  state_ = State::closed;
364  processWampReply(msg.requestKey(), std::move(msg));
365  }
366  else
367  {
368  auto reason = msg.fields.at(2).as<String>();
369  auto errc = lookupWampErrorUri(reason, SessionErrc::closeRealm);
370  abortPending(make_error_code(errc));
371  Message reply{ WampMsgType::goodbye,
372  {0u, Object{}, "wamp.error.goodbye_and_out"} };
373  send(reply);
374  state_ = State::closed;
375  }

393  bool checkValidMsg(WampMsgType type)
394  {
395  auto traits = MessageTraits::lookup(type);
396  bool valid = isMsgSupported(traits);
397 
398  if (!valid)
399  fail(ProtocolErrc::unsupportedMsg);
400  else
401  {
402  switch (state_)
403  {
404  case State::establishing:
405  valid = traits.forEstablishing;
406  break;
407 
408  case State::authenticating:
409  valid = traits.forChallenging;
410  break;
411 
412  case State::established:
413  case State::shuttingDown:
414  valid = traits.forEstablished;
415  break;
416 
417  default:
418  valid = false;
419  break;
420  }
421 
422  if (!valid)
423  fail(ProtocolErrc::unexpectedMsg);
424  }
425 
426  return valid;
427  }
