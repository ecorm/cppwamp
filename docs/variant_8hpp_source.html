926 template <typename T> Variant& Variant::operator=(T value)
927 {
928  static_assert(ArgTraits<T>::isValid, "Invalid argument type");
929 
930  using FieldType = typename ArgTraits<T>::FieldType;
931  if (is<FieldType>())
932  as<FieldType>() = std::move(value);
933  else
934  {
935  destruct();
936  constructAs<FieldType>(std::move(value));
937  typeId_ = FieldTraits<FieldType>::typeId;
938  }
939  return *this;
940 }
