29 template <typename T>
30 CPPWAMP_API void convert(FromVariantConverter& conv, std::unordered_set<T>& set)
31 {
32  const auto& variant = conv.variant();
33  if (!variant.is<Array>())
34  {
35  throw error::Conversion("Attempting to convert non-array variant "
36  "to std::unordered_set");
37  }
38 
39  std::unordered_set<T> newSet;
40  const auto& array = variant.as<Array>();
41  for (Array::size_type i=0; i<array.size(); ++i)
42  {
43  try
44  {
45  newSet.emplace(array[i].to<T>());
46  }
47  catch (const error::Conversion& e)
48  {
49  std::string msg = e.what();
50  msg += " (for element #" + std::to_string(i) + ")";
51  throw error::Conversion(msg);
52  }
53  }
54  set = std::move(newSet);
55 }
