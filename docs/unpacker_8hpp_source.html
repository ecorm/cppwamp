251 // This is caught internally by Client while dispatching RPCs and is never
252 // propagated through the API.

254 struct UnpackError : public Error
255 {
256  UnpackError() : Error("wamp.error.invalid_argument") {}
257 };
258 
259 //------------------------------------------------------------------------------
260 template <typename... A>
261 struct UnpackedArgGetter
262 {
263  template <int N>
264  static NthTypeOf<N, A...> get(const Array& args)
265  {
266  using TargetType = NthTypeOf<N, A...>;
267  try
268  {
269  return args.at(N).to<TargetType>();
270  }
271  catch (const error::Conversion& e)
272  {
273  std::ostringstream oss;
274  oss << "Type " << typeNameOf(args.at(N))
275  << " at arg index " << N
276  << " is not convertible to the RPC's target type";
277  throw UnpackError().withArgs(oss.str(), e.what());
278  }
279  }
280 };

295 //------------------------------------------------------------------------------
296 template <typename S, typename... A>
297 void EventUnpacker<S,A...>::operator()(Event&& event)
298 {
299  if (event.args().size() < sizeof...(A))
300  {
301  std::ostringstream oss;
302  oss << "Expected " << sizeof...(A)
303  << " args, but only got " << event.args().size();
304  throw internal::UnpackError().withArgs(oss.str());
305  }
306 
307  // Use the integer parameter pack technique shown in
308  // http://stackoverflow.com/a/7858971/245265
309  using Seq = typename internal::GenIntegerSequence<sizeof...(A)>::type;
310  invoke(std::move(event), Seq());
311 }
312 
313 //------------------------------------------------------------------------------
314 template <typename S, typename... A>
315 template <int ...Seq>
316 void EventUnpacker<S,A...>::invoke(Event&& event,
317  internal::IntegerSequence<Seq...>)
318 {
319  Array args = event.args();
320  using Getter = internal::UnpackedArgGetter<A...>;
321  slot_(std::move(event), Getter::template get<Seq>(args)...);
322 }
